## 版本说明
- V1.1：客户端已集成多种应用运行方式，包括
  - 页面级（vue页面，通过路由跳转和Router-link标签显示）
  - 网页级（任意框架的前端页面的链接，Iframe直接显示）
  - js程序级（任意js代码执行）
  - 可执行文件（任意可执行文件运行）
- V1.2：原生钱包和多钱包支持，包括
  - 集成内置浏览器miniBrowser，默认安装MetaMask钱包
  - 集成WalletConnect，支持钱包连接，全局调用

 > 说明，页面级支持wallectconnect钱包连接，网页级还支持metamask插件连接，但两者并不能互通

## 前置条件
1. 移动端安装MetaMask Mobile
   - 安卓端在[这里](https://github.com/MetaMask/metamask-mobile)下载安装
   - IOS的暂无，可使用WallectConnect官方在线的钱包[Demo](https://react-web3wallet.vercel.app/walletconnect)
2. 在移动端创建自己的私钥和账户
3. 领取测试网水龙头，以Goeril TestNet为例，请在[这里](https://goerlifaucet.com/)领取
4. 本次测试合约已部署好（USDT Constract address: 0x8942E52BE95b4f7e7566A91D77308F415DCf1d5D）

## 使用方法
1. 全局使用代码引入walletProvider对象
```javascript
import {
  useWeb3ModalProvider,
} from '@web3modal/ethers5/vue'
 let provider =  useWeb3ModalProvider();
 const walletProvider = provider.walletProvider.value;
```
2. 合约查询和调用的参考如下
```javascript
  <script lang="ts" setup>
  import {
  createWeb3Modal,
  defaultConfig,
  useWeb3Modal,
  useWeb3ModalEvents,
  useWeb3ModalProvider,
  useWeb3ModalAccount,
  useWeb3ModalState,
  useWeb3ModalTheme
} from '@web3modal/ethers5/vue'

  const projectId = '01c174cab89954d2942216e56549d410'
  if (!projectId) {
  throw new Error('VITE_PROJECT_ID is not set')
}

  // 2. Set chains
  const chains = [
  {
    chainId: 1,
    name: 'Ethereum',

    explorerUrl: 'https://etherscan.io',

  {

  },

    chainId: 5,
    name: 'Goerli',
    currency: 'ETH',
    explorerUrl: 'https://goerli.etherscan.io',
    rpcUrl: 'https://endpoints.omniatech.io/v1/eth/goerli/public'
  }
  ]

  const ethersConfig = defaultConfig({
  metadata: {
  name: 'Web3Modal',
  description: 'Web3Modal Laboratory',
  url: 'https://web3modal.com',

},

  projectId,
  chains,

  '--w3m-color-mix': '#00BB7F',
  '--w3m-color-mix-strength': 20
}
};

  console.log("WC配置：",options);
  let modal2 = createWeb3Modal(options);

  // 4. Use modal composable
  const modal = useWeb3Modal()  // 窗口控制工具
  const state = useWeb3ModalState()
  const { setThemeMode, themeMode, themeVariables } = useWeb3ModalTheme()
  const events = useWeb3ModalEvents()

  //  test events listener

  import { watch } from "vue";


  const web3ModalEvents = useWeb3ModalEvents();

  watch(web3ModalEvents, () => {
  console.log(web3ModalEvents.data);
  switch (web3ModalEvents.data.event) {
  case "MODAL_OPEN":
  console.log("测试事件响应：打开了窗口",);
  break;
  case "SELECT_WALLET":
  console.log("测试事件响应：选择了钱包",);
  break;
  case "CONNECT_SUCCESS":
  console.log("测试事件响应：连接成功",);
  case "DISCONNECT_SUCCESS":
  console.log("测试事件响应：断开成功",);
  break;
  case "CONNECT_ERROR":
  console.log("测试事件响应：连接失败",);
  break;
  default:
  break;
}
});

  // goerli testnet deployed usdt contract address
  const USDTAddress = '0x8942E52BE95b4f7e7566A91D77308F415DCf1d5D'

  // owner address
  const ownerAddress = '0x1423f4ceC7fCBE8400A957121ad616C439a0d4CF';
  // test address
  const testAddress = '0x437a5079CE5f89A62bc74e66e5f5a772Eb31CaB9';

  // The ERC-20 Contract ABI, which is a common contract interface
  // for tokens (this is the Human-Readable ABI format)
  const USDTAbi = [
  'function name() view returns (string)',
  'function symbol() view returns (string)',
  'function balanceOf(address) view returns (uint)',
  'function transfer(address to, uint amount)',
  'event Transfer(address indexed from, address indexed to, uint amount)'
  ]

  const localAccount = useWeb3ModalAccount();

  async function onQueryContract(address:string) {
  const walletProvider = modal2.getWalletProvider();
  if (!walletProvider) throw Error('User disconnected')

  const ethersProvider = new ethers.providers.Web3Provider(walletProvider)
  const signer = await ethersProvider.getSigner()
  // The Contract object
  const USDTContract = new ethers.Contract(USDTAddress, USDTAbi, signer)
  const USDTBalance = await USDTContract.balanceOf(address);
  console.log(USDTBalance.toNumber());
  // console.log(ethers.utils.formatUnits(USDTBalance, 18))
}


  async function onInvokeContract() {
  let localInstance =  useWeb3ModalProvider();
  const walletProvider = localInstance.walletProvider.value;
  console.log("两个provider实例：",modal2.getWalletProvider(),localInstance.walletProvider.value)
  if (!walletProvider) throw Error('User disconnected')

  const ethersProvider = new ethers.providers.Web3Provider(walletProvider)
  const signer = await ethersProvider.getSigner()
  // Query account first
  console.log("查询转账前账户余额：")
  await onQueryContract(ownerAddress);
  await onQueryContract(testAddress);
  // The Contract object
  console.log("执行转账：")
  const USDTContract = new ethers.Contract(USDTAddress, USDTAbi, signer)
  const invokeResult = await USDTContract.transfer(testAddress,ethers.BigNumber.from("10"));
  console.log("合约执行结果：",invokeResult)
  console.log("等待交易打包...")
  setTimeout(async function() {
  console.log("交易执行完成！");
  console.log("查询转账后账户余额：")
  await onQueryContract(ownerAddress);
  await onQueryContract(testAddress);
}, 10000);
}

  async function onQueryTransactions () {
  const walletProvider = modal2.getWalletProvider();
  if (!walletProvider) throw Error('User disconnected')

  const ethersProvider = new ethers.providers.Web3Provider(walletProvider)
  const signer = await ethersProvider.getSigner()
  // The Contract object
  // const publicKey = await signer.getPublickey();
  // console.log("钱包公钥：", publicKey);
  const etherscanProvider = new ethers.providers.EtherscanProvider();
  const transactionHistory = await etherscanProvider.getHistory(ownerAddress);
  console.log("钱包交易信息：",transactionHistory);
}

</script>

<template>

```
