```javascript
    url: tab.url,

  };
  await ipc.send('addTableTab', JSON.parse(JSON.stringify(args)));
```

## 🔄 完整流程

### 数据流向

```
用户点击图标
    ↓
icon.vue: newOpenApp()
    ↓ browser.openInTable(url, { wallet: true })
    ↓
browser.ts: openInTable()
    ├─ 检测 URL 类型（本地 HTML）
    ├─ 添加 wallet 参数到 URL
    ├─ 初始化钱包服务
    ├─ 连接钱包（如未连接）
    └─ 跳转到浏览器页面
    ↓
browser/Index.vue: mounted()
    ├─ 接收路由参数
    └─ 调用 invokeAddTab()
    ↓
browser/Index.vue: invokeAddTab()
    ├─ 准备参数（url, wallet, isLocalHtml）
    └─ 发送 IPC 消息 'addTableTab'
    ↓
tableTabManager.js: ipc.on('addTableTab')
    ├─ 接收消息
    └─ 调用 addTab()
    ↓
tableTabManager.js: addTab()
    ├─ 检测 isLocalHtml
    ├─ 使用 render.getUrl() 转换 URL
    ├─ 创建 BrowserView
    ├─ 加载页面
    └─ 发送 'init-wallet' 消息（如启用钱包）
    ↓
wallet-test.html
    ├─ 页面加载
    ├─ 环境检测
    ├─ 读取 URL 参数
    ├─ 监听 'init-wallet' 消息
    └─ 自动初始化钱包（如参数为 true）
```

### 2. 参数传递

**问题**: 如何将钱包参数传递到新窗口？

**解决**: 通过 URL 参数 + IPC 消息

```javascript
// URL 参数

// IPC 消息
```

### 3. 钱包服务访问

**问题**: 内嵌页面如何访问钱包服务？

**解决**: 通过 `window.wallet` 全局对象

```javascript
// 在 HTML 中
if (typeof window.wallet !== 'undefined') {
  await window.wallet.initWallet();
}
```

## 🐛 已知问题和解决方案

### 问题 1: 钱包服务未注入

**原因**: 内嵌页面的 webPreferences 配置问题

**解决**: 确保 `contextIsolation: true` 且有正确的 preload

---

### 问题 2: URL 参数丢失

**原因**: 参数在传递过程中被过滤

**解决**: 在每个环节都保留完整的 URL

---

### 问题 3: 页面加载失败

**原因**: 本地文件路径未正确转换

**解决**: 使用 `render.getUrl()` 转换
