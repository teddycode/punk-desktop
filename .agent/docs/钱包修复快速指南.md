### 第 2 步：在桌面主窗口添加 IPC 监听器

#### 2.1 确定监听器添加位置

有两个选择：
- **选项 A：** 在 `Browser/Index.vue` 中（推荐 - 与浏览器逻辑耦合）
- **选项 B：** 在 `Splash.vue` 中（适合全局钱包管理）

#### 2.2 添加监听器代码

**文件：** `vite/packages/table/page/Browser/Index.vue`

找到 `setup()` 或 `mounted()` 部分，添加：

```vue
<script setup>
import { inject, onMounted, onUnmounted } from 'vue';

// 获取钱包服务（由 Splash.vue 提供）
const wallet = inject('wallet');

// IPC 监听器
const setupWalletIPC = () => {
  console.log('[Browser] 设置钱包 IPC 监听器');
  
  // 1. 钱包初始化请求
  window.ipc.on('wallet-init-request', async (data) => {
    console.log('[Browser] 收到钱包初始化请求', data);
    try {
      const initialized = wallet?.checkInitialized?.() || false;
      console.log('[Browser] 钱包初始化状态:', initialized);
      
      // 发送响应回主进程
      window.ipc.send('wallet-init-complete', {
        success: true,
        initialized: initialized
      });
    } catch (error) {
      console.error('[Browser] 钱包初始化错误:', error);
      window.ipc.send('wallet-init-complete', {
        success: false,
        error: error.message
      });
    }
  });

  // 2. 钱包连接请求
  window.ipc.on('wallet-connect-request', async (data) => {
    console.log('[Browser] 收到钱包连接请求', data);
    try {
      if (!wallet) {
        throw new Error('钱包服务未注入');
      }
      
      // 调用桌面钱包服务
      const account = await wallet.openWallet();
      console.log('[Browser] 钱包连接成功:', account);
      
      window.ipc.send('wallet-connect-complete', {
        success: true,
        address: wallet.address?.value || null,
        chainId: wallet.chainId?.value || null
      });
    } catch (error) {
      console.error('[Browser] 钱包连接错误:', error);
      window.ipc.send('wallet-connect-complete', {
        success: false,
        error: error.message
      });
    }
  });

  // 3. 钱包断开请求
  window.ipc.on('wallet-disconnect-request', async (data) => {
    console.log('[Browser] 收到钱包断开请求', data);
    try {
      if (!wallet) {
        throw new Error('钱包服务未注入');
      }
      
      await wallet.disconnectWallet();
      console.log('[Browser] 钱包已断开');
      
      window.ipc.send('wallet-disconnect-complete', {
        success: true
      });
    } catch (error) {
      console.error('[Browser] 钱包断开错误:', error);
      window.ipc.send('wallet-disconnect-complete', {
        success: false,
        error: error.message
      });
    }
  });

  // 4. 获取钱包状态请求
  window.ipc.on('wallet-get-state-request', async (data) => {
    console.log('[Browser] 收到获取钱包状态请求', data);
    try {
      const state = {
        success: true,
        initialized: wallet?.checkInitialized?.() || false,
        connected: wallet?.isConnected?.value || false,
        address: wallet?.address?.value || null,
        chainId: wallet?.chainId?.value || null
      };
      console.log('[Browser] 钱包状态:', state);
      
      window.ipc.send('wallet-get-state-complete', state);
    } catch (error) {
      console.error('[Browser] 获取钱包状态错误:', error);
      window.ipc.send('wallet-get-state-complete', {
        success: false,
        error: error.message
      });
    }
  });
};

// 清理监听器
const cleanupWalletIPC = () => {
  console.log('[Browser] 清理钱包 IPC 监听器');
  window.ipc?.removeAllListeners?.('wallet-init-request');
  window.ipc?.removeAllListeners?.('wallet-connect-request');
  window.ipc?.removeAllListeners?.('wallet-disconnect-request');
  window.ipc?.removeAllListeners?.('wallet-get-state-request');
};

// 生命周期钩子
onMounted(() => {
  setupWalletIPC();
});

onUnmounted(() => {
  cleanupWalletIPC();
});
</script>
```

**如果是 Options API 写法（不是 setup）：**

    cleanupWalletIPC() {
      console.log('[Browser] 清理钱包 IPC 监听器');

### 第 3 步：完善主进程的响应转发

#### 3.1 修改 tableTabManager.js

**文件：** `src/main/tableTabManager.js`

找到 `ipc.handle('wallet-init')` 等处理器，修改为等待响应：

/**
 * 钱包初始化
 */
ipc.handle('wallet-init', async (event) => {
  console.log('[TableTabManager IPC] wallet-init 调用');
  try {
    if (!this.tableWin || this.tableWin.isDestroyed()) {
      return { success: false, error: '桌面窗口不可用' };
    }
    
    // 创建 Promise 等待响应
    return new Promise((resolve, reject) => {
      const timeout = setTimeout(() => {
        ipc.off('wallet-init-complete', listener);
        reject(new Error('钱包初始化超时'));
      }, 10000); // 10秒超时

      // 监听响应
      const listener = (e, result) => {
        clearTimeout(timeout);
        ipc.off('wallet-init-complete', listener);
        console.log('[TableTabManager IPC] wallet-init 响应:', result);
        resolve(result);
      };
      
      ipc.once('wallet-init-complete', listener);
      
      // 发送请求到桌面
      this.tableWin.webContents.send('wallet-init-request', {
        requestId: Date.now()
      });
    });
  } catch (error) {
    console.error('[TableTabManager IPC] wallet-init 错误:', error);
    return { 
      success: false, 
      error: error.message 
    };
  }
});

/**
 * 钱包连接
 */
ipc.handle('wallet-connect', async (event) => {
  console.log('[TableTabManager IPC] wallet-connect 调用');

## 🐛 故障排查

### 问题 1：window.wallet 是 undefined
**原因：** Preload 脚本未加载或路径错误

**检查：**
1. 确认文件存在：`ls js/preload/walletPreload.js`
2. 检查主进程日志是否有 `[TableTabManager] 🔐 钱包功能已启用`
3. 检查 DevTools Console 是否有 preload 错误

**解决：**
- 检查 tableTabManager.js 中 preload 路径是否正确
- 确保 `wallet: true` 参数正确传递

---

### 问题 2：IPC 调用超时
**原因：** 桌面 IPC 监听器未设置或 ipc.send 的通道名不匹配

**检查：**
1. Browser/Index.vue 中是否添加了 `setupWalletIPC()`
2. `ipc.on` 的通道名是否与 `webContents.send` 一致
   - 请求：`wallet-init-request`
   - 响应：`wallet-init-complete`
3. 渲染进程是否重新构建 (`cd vite && yarn run build`)

**解决：**
- 在 Browser/Index.vue 的 `mounted()` 中添加日志：
  ```javascript
  console.log('[Browser] setupWalletIPC 已调用');
  ```
- 确认日志出现后再测试

---

### 问题 3：钱包服务报错 "未注入"
**原因：** Splash.vue 中的钱包服务未通过 `provide` 提供

**检查：**
查看 Splash.vue 是否有：
```javascript
import { provide } from 'vue';
import { useWalletService } from './composables/useWalletService';

setup() {
  const wallet = useWalletService();
  provide('wallet', wallet);
  // ...
}
```

**解决：**
- 如果没有 `provide('wallet', ...)`，添加它
- 或者在 Browser/Index.vue 中直接导入钱包服务
